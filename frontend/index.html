async function generateImage() { const prompt = document.getElementById("prompt").value; const imageEl = document.getElementById("image"); imageEl.src = ""; // נקה תוצאה קודמת if (!prompt.trim()) { alert("אנא הזן תיאור לתמונה."); return; } try { const response = await fetch("https://ai-alpha-azure.vercel.app/generate", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ prompt }) }); const data = await response.json(); if (data.image) { imageEl.src = data.image; } else { alert("❌ לא התקבלה תמונה. נסה שוב."); console.error("תשובת שרת:", data); } } catch (error) { alert("🚫 שגיאה בעת יצירת התמונה."); console.error("שגיאה:", error); } }
